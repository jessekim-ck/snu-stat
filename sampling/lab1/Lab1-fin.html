Neyman vs Proportional allocation.

#-----------------------------------------------------------------------------------------#
#-----------------------------------------------------------------------------------------#
#----------------------------- Sample allocation    --------------------------------------#
#-----------------------------------------------------------------------------------------#
#-----------------------------------------------------------------------------------------#


N_h<-c()
true.mean.wage_h<-c() # true mean wage per area
true.var.wage_h<-c() # true var wage per area
for(i in 1 : length(data.exaple)){
  N_h<-c(N_h,dim(data.exaple[[i]])[1])  # 각 층의 모집단 수
  true.mean.wage_h<-c(true.mean.wage_h,mean(data.exaple[[i]][,3]) ) # 각 층의 임금 평균 참값 
  true.var.wage_h<-c(true.var.wage_h,var(data.exaple[[i]][,3])*(dim(data.exaple[[i]])[1]-1)/dim(data.exaple[[i]])[1] ) # 각 층의 임금 분산 참값
}
N<-sum(N_h)
D=1 # 최대 오차 허용한계 = +-2 -> D=2^2/4=1

# Neyman allocation (c_h=c, c_h : 비용함수 )
n_neyman<-round((sum(N_h*sqrt(true.var.wage_h))^2)/(N^2*D+sum(N_h*true.var.wage_h))) # Neyman allocation total sample number
n_neyman_h<-round((N_h*sqrt(true.var.wage_h))/(sum(N_h*sqrt(true.var.wage_h)))*n_neyman) # Neyman allocation sample number per area

# proportional allocation (n_h = c*N_h, c= constant)
n_prop<-round((sum(N_h*true.var.wage_h))/(N*D+sum(N_h*true.var.wage_h)/N))# Proortion allocation total sample number
n_prop_h<-round((N_h/N)*n_prop)# Proortion allocation sample number per area


# Comparision : Neyman and SRS
# We approximate MSE using 1000 iterations
wage.est.srs<-c()
wage.est.neyman<-c()
for(j in 1 : 1000){
  wage.sample.neyman<-c()
  for(i in 1: 10){
    wage.sample.neyman<-c(wage.sample.neyman, mean(sample(data.exaple[[i]][,3],n_prop_h[i],replace=F) ) ) 
  }
  wage.est.neyman[j]<-sum(wage.sample.neyman*N_h/N)
  
  wage.sample.srs<-sample(wage.vector,n_neyman,replace=F) 
  wage.est.srs[j]<-mean(wage.sample.srs)
}

cat("Neyman allocation MSE is : " ,mean((wage.est.neyman-true.mean.wage)^2))
## Neyman allocation MSE is :  1.037737
cat("SRS MSE is : " ,mean((wage.est.srs-true.mean.wage)^2))
## SRS MSE is :  2.770593
cat("MSE ratio of the Neyman allocation to SRS is :"  ,mean((wage.est.neyman-true.mean.wage)^2)/ mean((wage.est.srs-true.mean.wage)^2))
## MSE ratio of the Neyman allocation to SRS is : 0.3745542
# Comparision : Prop and SRS
# We approximate MSE using 1000 iterations
wage.est.srs<-c()
wage.est.prop<-c()
for(j in 1 : 1000){
  wage.sample.prop<-c()
  for(i in 1: 10){
    wage.sample.prop<-c(wage.sample.prop, mean(sample(data.exaple[[i]][,3],n_prop_h[i],replace=F) ) ) 
  }
  wage.est.prop[j]<-sum(wage.sample.prop*N_h/N)
  
  wage.sample.srs<-sample(wage.vector,n_prop,replace=F) 
  wage.est.srs[j]<-mean(wage.sample.srs)
}

cat("Proportion allocation MSE is : " ,mean((wage.est.prop-true.mean.wage)^2))
## Proportion allocation MSE is :  0.9773597
cat("SRS MSE is : " ,mean((wage.est.srs-true.mean.wage)^2))
## SRS MSE is :  1.19223

## MSE ratio of the Proportion allocation to SRS is : 0.8197745

# We approximate MSE using 1000 iterations
n.post=(n_neyman+n_prop)/2 # we choose total sample number by (n_neyman+n_prop)/2
wage.est.srs<-c()
wage.est.post<-c()  
n.post_h<-c()
wage.est.post_h<-c()
for(j in 1 : 1000){
  sample_index<-sample(1:N,n.post,replace=F)
  wage.vector.sample<-wage.vector[sample_index]
  wage.est.srs[j]<-mean(wage.vector.sample) # SRS
  
  for( i in 1 : length(data.exaple)){
    n.post_h[i]<-sum(Area[sample_index]==paste("A",i,sep="")) 
    wage.est.post_h[i]<-mean(wage.vector.sample[Area[sample_index]==paste("A",i,sep="")])
  }
  
  wage.est.post[j]<-sum(N_h/N*wage.est.post_h)
  
  
}
